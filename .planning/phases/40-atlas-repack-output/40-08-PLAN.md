     * Phase 40 D-07 — SHA256 baseline refresh for repack regression tests.
     *
     * Regenerates:
     *   - tests/fixtures/repack-baselines.json
     *   - tests/fixtures/repack-expected/SIMPLE_TEST.atlas

     *
     * Build dependency: requires `npm run build` first so the worker modules
     * are emitted to out/main/*.js. If running pre-build, error loudly.
     */

    import path from 'node:path';
    import os from 'node:os';
    import crypto from 'node:crypto';
    import { fileURLToPath } from 'node:url';
    import { createRequire } from 'node:module';

    const __dirname = path.dirname(__filename);
    const REPO_ROOT = path.resolve(__dirname, '..');

    const EXPECTED_ATLAS_DIR = path.join(REPO_ROOT, 'tests', 'fixtures', 'repack-expected');

    const FIXTURES = [
      {
        name: 'SIMPLE_TEST',
        skeletonPath: 'fixtures/SIMPLE_PROJECT/SIMPLE_TEST.json',
        atlasPath: 'fixtures/SIMPLE_PROJECT/SIMPLE_TEST.atlas',
      },
    ];

    function sha256(buf) {
      return crypto.createHash('sha256').update(buf).digest('hex');
    }

    async function refreshFixture(fixture) {
      const tmpDir = await fs.mkdtemp(path.join(os.tmpdir(), 'stm-repack-refresh-'));
      try {
        let runRepack, runExport;
        try {

            'repack-refresh-baselines: workers not found at out/main/. Run `npm run build` first, then re-invoke.',
          );
        }

        // Plan shape MUST match the test files' plan shape exactly — otherwise
        // baselines won't match real outputs.

        const plan = {
          rows: [
            { sourcePath, outPath: 'images/CIRCLE.png',   sourceW: 699, sourceH: 699, outW: 699, outH: 699, effectiveScale: 1.0, attachmentNames: ['CIRCLE'] },
            { sourcePath, outPath: 'images/SQUARE.png',   sourceW: 1000, sourceH: 1000, outW: 1000, outH: 1000, effectiveScale: 1.0, attachmentNames: ['SQUARE'] },
            { sourcePath, outPath: 'images/TRIANGLE.png', sourceW: 833, sourceH: 759, outW: 833, outH: 759, effectiveScale: 1.0, attachmentNames: ['TRIANGLE'] },
          ],
          excludedUnused: [],
          passthroughCopies: [],
          totals: { count: 3 },
        };

        const looseWritten = new Set();
        await runExport(plan, tmpDir, () => {}, () => false, true, false, looseWritten);

        const loose = {};
        for (const row of plan.rows) {
          const p = path.join(tmpDir, row.outPath);
          const buf = await fs.readFile(p);
          loose[row.outPath] = sha256(buf);
        }

        const atlasDir = path.join(tmpDir, 'atlas');
        await fs.mkdir(atlasDir, { recursive: true });
        const atlasWritten = new Set();
        const repackResult = await runRepack(
          plan, atlasDir, () => {}, () => false, true, false,
          { maxPageSize: 4096, allowRotation: false, padding: 2 },
          atlasWritten,
        );

        const atlas = {};
        const atlasBuf = await fs.readFile(repackResult.atlasFile);
        atlas[path.basename(repackResult.atlasFile)] = sha256(atlasBuf);
        for (const p of repackResult.pageFiles) {
          const buf = await fs.readFile(p);
          atlas[path.basename(p)] = sha256(buf);
        }

        await fs.mkdir(EXPECTED_ATLAS_DIR, { recursive: true });
        await fs.writeFile(path.join(EXPECTED_ATLAS_DIR, `${fixture.name}.atlas`), atlasBuf, 'utf8');

        return { loose, atlas };
      } finally {
        await fs.rm(tmpDir, { recursive: true, force: true });
      }
    }

      console.log('repack-refresh-baselines: regenerating SHA256 baselines...');

      const require = createRequire(import.meta.url);
      const sharpPkg = require(path.join(REPO_ROOT, 'node_modules/sharp/package.json'));

        },

        out[fixture.name] = await refreshFixture(fixture);
      }

      await fs.mkdir(path.dirname(BASELINE_PATH), { recursive: true });
      await fs.writeFile(BASELINE_PATH, JSON.stringify(out, null, 2) + '\n', 'utf8');
      console.log(`repack-refresh-baselines: wrote ${BASELINE_PATH}`);
      console.log(`repack-refresh-baselines: wrote ${EXPECTED_ATLAS_DIR}/SIMPLE_TEST.atlas`);
      console.log('Done. Commit the regenerated files alongside any dep bump that triggered the refresh.');
    }

    main().catch((err) => {
      console.error('repack-refresh-baselines: FAILED');
      console.error(err);
      process.exit(1);
    });

    function loadBaselines(): { _meta?: Record<string, string>; SIMPLE_TEST?: { loose?: Record<string, string>; atlas?: Record<string, string> } } {

        );
      }
      return JSON.parse(fs.readFileSync(BASELINE_PATH, 'utf8'));
    }

    function saveBaselines(b: unknown) {
      fs.writeFileSync(BASELINE_PATH, JSON.stringify(b, null, 2) + '\n', 'utf8');
    }
