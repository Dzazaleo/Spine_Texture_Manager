       ```typescript
       export function validateRecentFile(
         input: unknown,
       ): { ok: true; file: RecentFileV1 } | { ok: false } {
         if (!input || typeof input !== 'object') return { ok: false };
         const obj = input as Record<string, unknown>;

         // Version FIRST — gate before any other field interpretation (mirrors
         // validateProjectFile lines 96-120; D-151 / D-156 discipline carried
         // forward; per D-177 unknown OR newer versions both load as []).
         if (typeof obj.version !== 'number') return { ok: false };
         if (obj.version !== 1) return { ok: false };

         if (!Array.isArray(obj.paths)) return { ok: false };
         for (const p of obj.paths) {
           if (typeof p !== 'string' || p.length === 0) return { ok: false };
         }
         return { ok: true, file: { version: 1, paths: obj.paths as string[] } };
       }
       ```

    4. **writeRecentFileAtomic helper** (lift PATTERNS.md lines 49-58 verbatim — Pitfall 2 same-directory tmp; 2-space JSON for human-readable debugging):
       ```typescript
       async function writeRecentFileAtomic(
         recent: RecentFileV1,
         finalPath: string,
       ): Promise<void> {
         const json = JSON.stringify(recent, null, 2);
         const tmpPath = finalPath + '.tmp';
         await writeFile(tmpPath, json, 'utf8');
         await rename(tmpPath, finalPath);
       }
       ```

    5. **Public surface** (lift PATTERNS.md lines 101-124 verbatim):
       ```typescript
       export async function loadRecent(): Promise<string[]> {
         try {
           const text = await readFile(RECENT_PATH, 'utf8');
           const parsed: unknown = JSON.parse(text);
           const v = validateRecentFile(parsed);
           return v.ok ? v.file.paths : [];
         } catch {
           return [];
         }
       }

       export async function addRecent(absolutePath: string): Promise<string[]> {
         const current = await loadRecent();
         const filtered = current.filter((p) => p !== absolutePath);
         const next = [absolutePath, ...filtered].slice(0, MAX_RECENT);
         await writeRecentFileAtomic({ version: 1, paths: next }, RECENT_PATH);
         return next;
       }

       export async function clearRecent(): Promise<void> {
         await writeRecentFileAtomic({ version: 1, paths: [] }, RECENT_PATH);
       }
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| recent.json on disk → main process | Untrusted input crosses here every menu rebuild — a malicious or corrupted recent.json must NOT crash the app or expose unexpected paths to the menu builder. |
| addRecent caller → recent.ts | The absolutePath argument is supplied by main-side callers (project-io after successful Save As / Open). recent.ts itself does no path validation — it trusts callers (Plan 03 wires-in only verified absolute paths). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08.2-01-01 | Tampering | recent.json on disk (malicious payload, e.g. `{"version":1,"paths":["__proto__"]}`) | mitigate | `validateRecentFile` rejects non-string elements via per-element `typeof p !== 'string'` check; the validator gates on a literal `version === 1` (no string-keyed obj.proto traversal). Caller sees `[]` on any rejection. Verified by case (h). |
| T-08.2-01-02 | Tampering | recent.json with non-array paths (`{"version":1,"paths":"oops"}`) | mitigate | `Array.isArray(obj.paths)` gate. Verified by case (h). |
| T-08.2-01-03 | DoS | recent.json with extreme-size paths array | accept | Cap at 10 entries on every write (MAX_RECENT). A pre-existing 1M-entry recent.json would still be loaded into memory once before being filtered down on the next `addRecent`, but this requires file-system write access, at which point the threat model has already failed. Practical risk: low. |
| T-08.2-01-04 | Information Disclosure | recent.json reveals filesystem paths | accept | Recent file paths are user-chosen workspace artifacts; this is the same exposure as the macOS NSDocumentController standard-recent-files store. No secrets pass through this list. |
| T-08.2-01-05 | Repudiation | corrupt recent.json silently loads as [] without surfacing the error | accept | Per D-177 explicit decision — recent is non-critical UX state and must not block app launch. The diagnostic signal is "no recent appears in the menu." A future Phase 9 polish could log to console; out of scope here. |
| T-08.2-01-06 | Spoofing (caller side) | A future bug in Plan 03 calls `addRecent('/etc/passwd')` (non-stmproj path) | accept | recent.ts is a dumb store; path validity is a Plan 03 responsibility (only call after successful Save As / Open). Reasoning: defense-in-depth in recent.ts would require schema-validating each path's extension, but `Open Recent` clicking on a stale path already routes through `handleProjectOpenFromPath` which has its own .stmproj-extension validation (Phase 8 D-145 + project-io guards). |
