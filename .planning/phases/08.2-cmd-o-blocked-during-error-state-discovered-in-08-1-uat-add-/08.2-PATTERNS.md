#### Atomic-write pattern (lift from `src/main/project-io.ts:172-221`)

```typescript
async function writeRecentFileAtomic(
  recent: RecentFileV1,
  finalPath: string,
): Promise<void> {
  const json = JSON.stringify(recent, null, 2);
  const tmpPath = finalPath + '.tmp';
  await writeFile(tmpPath, json, 'utf8');
  await rename(tmpPath, finalPath);
}
```

**Notes:**
- Same-directory `<path>.tmp` (Pitfall 2 — avoids EXDEV cross-device errors).
- `writeFile` then `rename` (POSIX atomic; Windows best-effort, acceptable per project-io comment lines 165-171).
- 2-space indent matches project-io.ts:180 — keeps recent.json human-readable for debugging.
- Unlike `writeProjectFileAtomic`, `recent.ts` does NOT return a `SaveResponse` envelope. Per D-177 "non-critical UX state and must not block app launch" — failures are swallowed by callers (return `[]` from `loadRecent`).

#### Hand-rolled type guard pattern (lift from `src/core/project-file.ts:84-120`)

```typescript
type RecentFileV1 = { version: 1; paths: string[] };

export function validateRecentFile(
  input: unknown,
): { ok: true; file: RecentFileV1 } | { ok: false } {
  if (!input || typeof input !== 'object') return { ok: false };
  const obj = input as Record<string, unknown>;

  // Version FIRST — gate before any other field interpretation (mirrors
  // validateProjectFile lines 96-120; D-151 discipline carried forward).
  if (typeof obj.version !== 'number') return { ok: false };
  if (obj.version !== 1) return { ok: false }; // unknown OR newer — both load as []

  if (!Array.isArray(obj.paths)) return { ok: false };
  for (const p of obj.paths) {
    if (typeof p !== 'string' || p.length === 0) return { ok: false };
  }
  return { ok: true, file: { version: 1, paths: obj.paths as string[] } };
}
```

**Differences vs `validateProjectFile`:**
- The result envelope is simpler (just `{ ok: false }` — no error.kind/message). Per D-177, recent.json failures are silent — the renderer never sees a typed error.
- No "newer-version" carve-out: future schema bumps are also silent (load as `[]`). This is intentional — recent.json is non-critical.

#### Public surface (per CONTEXT §domain "Exports")

```typescript
const RECENT_PATH = path.join(app.getPath('userData'), 'recent.json');
const MAX_RECENT = 10;

export async function loadRecent(): Promise<string[]> {
  try {
    const text = await readFile(RECENT_PATH, 'utf8');
    const parsed: unknown = JSON.parse(text);
    const v = validateRecentFile(parsed);
    return v.ok ? v.file.paths : [];
  } catch {
    return [];
  }
}

export async function addRecent(absolutePath: string): Promise<string[]> {
  const current = await loadRecent();
  // Dedupe + push-to-front + cap at 10.
  const filtered = current.filter((p) => p !== absolutePath);
  const next = [absolutePath, ...filtered].slice(0, MAX_RECENT);
  await writeRecentFileAtomic({ version: 1, paths: next }, RECENT_PATH);
  return next;
}

export async function clearRecent(): Promise<void> {
  await writeRecentFileAtomic({ version: 1, paths: [] }, RECENT_PATH);
}
```

#### Pattern (lift from Electron `Menu.buildFromTemplate` + canonical_refs §Electron API)

```typescript
import { Menu, type MenuItemConstructorOptions } from 'electron';
import { loadRecent, clearRecent } from './recent.js';

export type MenuState = {
  canSave: boolean;
  canSaveAs: boolean;
  modalOpen: boolean;
};

export async function buildAppMenu(
  state: MenuState,
  mainWindow: Electron.BrowserWindow | null,
): Promise<Electron.Menu> {
  const recent = await loadRecent();
  const fileDisabled = state.modalOpen;

  const recentSubmenu: MenuItemConstructorOptions[] =
    recent.length === 0
      ? [{ label: '(No recent projects)', enabled: false }]
      : recent.map((p) => ({
          label: p.split(/[\\/]/).pop() ?? p,
          toolTip: p, // Discretion: full absolute path on hover
          enabled: !fileDisabled,
          click: () => mainWindow?.webContents.send('menu:open-recent-clicked', p),
        }));
  recentSubmenu.push({ type: 'separator' });
  recentSubmenu.push({
    label: 'Clear Menu',
    enabled: recent.length > 0,
    click: async () => {
      await clearRecent();
      // Caller (applyMenu) re-installs after the click resolves.
    },
  });

  const template: MenuItemConstructorOptions[] = [
    { role: 'appMenu' }, // D-188: macOS App-name with About / Hide / Quit
    {
      label: 'File',
      submenu: [
        {
          label: 'Open…',
          accelerator: 'CmdOrCtrl+O',
          enabled: !fileDisabled,
          click: () => mainWindow?.webContents.send('menu:open-clicked'),
        },
        { label: 'Open Recent', submenu: recentSubmenu },
        { type: 'separator' },
        {
          label: 'Save',
          accelerator: 'CmdOrCtrl+S',
          enabled: state.canSave && !fileDisabled,
          click: () => mainWindow?.webContents.send('menu:save-clicked'),
        },
        {
          label: 'Save As…',
          accelerator: 'CmdOrCtrl+Shift+S',
          enabled: state.canSaveAs && !fileDisabled,
          click: () => mainWindow?.webContents.send('menu:save-as-clicked'),
        },
        { type: 'separator' },
        { role: 'close' }, // Cmd+W via role
      ],
    },
    { role: 'editMenu' }, // D-188: free Cut/Copy/Paste/SelectAll for text inputs
    { role: 'viewMenu' }, // D-188: includes Reload/DevTools — DevTools auto-hidden when packaged per Electron's role behaviour
    { role: 'windowMenu' }, // D-188
    { role: 'help', submenu: [] }, // Phase 9 may add docs links
  ];

  return Menu.buildFromTemplate(template);
}

export async function applyMenu(
  state: MenuState,
  mainWindow: Electron.BrowserWindow | null,
): Promise<void> {
  const menu = await buildAppMenu(state, mainWindow);
  Menu.setApplicationMenu(menu);
}
```

**Discretion calls preserved (per CONTEXT §"Claude's Discretion"):**
- Open Recent label = basename + `toolTip: full path`.
- Empty submenu = single disabled `(No recent projects)` placeholder + separator + disabled Clear Menu.
- IPC channel naming = per-action (`menu:open-clicked`, `menu:open-recent-clicked`, `menu:save-clicked`, `menu:save-as-clicked`).

## Shared Patterns

### Pattern A: Atomic-write idiom (settings file)

**Source:** `src/main/project-io.ts:172-221` (`writeProjectFileAtomic`) — closest analog. Original pattern: `src/main/image-worker.ts:255-304` (per-row PNG write).

**Apply to:** `src/main/recent.ts` (`writeRecentFileAtomic`).

```typescript
const tmpPath = finalPath + '.tmp';
await writeFile(tmpPath, json, 'utf8');
await rename(tmpPath, finalPath);
```

**Pitfall 2:** same-directory tmp avoids EXDEV cross-device errors. POSIX rename is atomic; Windows is best-effort (acceptable for non-critical settings).

---

### Pattern B: Hand-rolled type guard with discriminated envelope

**Source:** `src/core/project-file.ts:84-203` (`validateProjectFile`).

**Apply to:** `src/main/recent.ts` (`validateRecentFile`).

Key discipline:
- Version field checked FIRST, before any other field interpretation.
- Per-field shape checks (no schema library).
- Discriminated `{ ok: true; ... } | { ok: false; ... }` envelope — caller narrows via `if (result.ok)`.

---

### Pattern C: One-way fire-and-forget IPC (`ipcMain.on` + `ipcRenderer.send`)

**Source:** `src/main/ipc.ts:496-500` (`'export:cancel'`); `src/main/index.ts:91-97` (`'project:confirm-quit-proceed'`); preload `src/preload/index.ts:113-115` (`cancelExport`), `:237-239` (`confirmQuitProceed`).

**Apply to:** `'menu:notify-state'` (renderer → main) and the four `'menu:*-clicked'` channels (main → renderer via `webContents.send`).

```typescript
// Main:
ipcMain.on('menu:notify-state', (_evt, state) => { /* validate, mutate, applyMenu */ });
// Preload:
notifyMenuState: (state) => { ipcRenderer.send('menu:notify-state', state); },
```

**NEVER use `ipcMain.handle` here** — these are one-way; the main-side handler returns void.

---

### Pattern D: Listener-identity preservation (Pitfall 9)

**Source:** `src/preload/index.ts:126-132` (`onExportProgress`); `:220-226` (`onCheckDirtyBeforeQuit`).

**Apply to:** all four `onMenu*` preload methods.

```typescript
const wrapped = (_evt, ...args) => handler(...args); // capture BEFORE .on
ipcRenderer.on(channel, wrapped);
return () => { ipcRenderer.removeListener(channel, wrapped); };
```

`ipcRenderer.removeListener` compares by reference. Anonymous wrappers in the unsubscribe closure silently leak listeners.

---

### Pattern E: Imperative ref-bridge (App.tsx ↔ AppShell)

**Source:** `src/renderer/src/App.tsx:178-188` (`beforeDropRef` declaration); `src/renderer/src/components/AppShell.tsx:678-702` (the registration `useEffect`).

**Apply to:** `appShellMenuRef` for menu-driven Save/Save As routing.

Same shape:
- App.tsx owns the `useRef<{ ... } | null>(null)` and passes the ref as a prop.
- AppShell registers `appShellMenuRef.current = { ... }` in a `useEffect`.
- Cleanup nulls the ref on unmount.
- App.tsx's menu-event handler dereferences the ref at call time (`appShellMenuRef.current?.onClickSave()`) so it always sees the latest registered impl. When AppShell is unmounted, the ref is null and the call is a no-op.

---

### Pattern F: Recovery-banner / no-AppShell menu-state push

**Source:** the existing pattern at `src/renderer/src/App.tsx:241-278` (the projectLoadFailed branch UI) — NOT a previous `notifyMenuState` push (this is brand new). The "applies in idle/error/projectLoadFailed states" predicate parallels the existing branching at lines 207-289.

**Apply to:** App.tsx's new `useEffect` keyed on `state.status` that pushes `{ canSave: false, canSaveAs: false, modalOpen: false }` when `status` is `idle` / `error` / `projectLoadFailed`.

This is what makes Cmd+O work in the recovery banner state — the menu accelerator stays enabled because main's File→Open keeps `enabled: true` while `modalOpen: false`. **This is the entire 08.1 UAT bug fix.**
