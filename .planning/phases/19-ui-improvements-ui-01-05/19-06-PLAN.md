const totalUsedFiles = props.plan.rows.length;
const toResize = props.plan.rows.filter((r) => r.outW < r.sourceW).length;
const sumSourcePixels = props.plan.rows.reduce(
  (acc, r) => acc + r.sourceW * r.sourceH,

  0,
);
