    import { describe, it, expect } from 'vitest';
    import * as fs from 'node:fs';
    import { Bone } from '@esotericsoftware/spine-core';
    import { loadSkeleton } from '../../src/core/loader.js';
    import { sampleSkeleton, DEFAULT_SAMPLING_HZ } from '../../src/core/sampler.js';

    const FIXTURE = 'fixtures/SIMPLE_PROJECT/SIMPLE_TEST.json';

    describe('sampler (N1.1–N1.6, N2.1, N2.3)', () => {
      it('N1.1+N1.2: returns peak records for every textured attachment in the default skin', () => {
        const load = loadSkeleton(FIXTURE);
        const peaks = sampleSkeleton(load);
        // Expect CIRCLE, SQUARE, SQUARE2 (shares SQUARE region), TRIANGLE. PATH is skipped.

        // Fixture inspection: CIRCLE mesh vertex 0 has bones=[4,5], weights=[0.00761, 0.99239].
        // Bone index 5 is the DOMINANT weight for the first vertex. If we inflate that bone's
        // scale pre-sample, the mesh vertex should move proportionally, and the mesh's peak
        // worldW should grow materially. If the sampler ignored weights, the peak wouldn't change.
        //
        // Strategy: load the fixture twice — once vanilla, once with bone[5].scale.* doubled
        // via a mutation hook on the SkeletonData's bone setup pose. Compare peak worldW.
        //
        // We mutate the SkeletonData's bone setup transform BEFORE instantiating Skeleton,
        // so every sampler-internal setToSetupPose() picks up the mutated scale.

        const baseCircle = [...baseline.values()].find(r => r.attachmentName === 'CIRCLE');

        // A tolerance of 1e-6 is tight enough to fail on any broken constraint wiring while
