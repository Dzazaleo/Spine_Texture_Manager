```typescript
  data: SlotData;       // data.name, data.index
  getAttachment(): Attachment | null;
  deform: number[];     // populated by DeformTimeline
  sequenceIndex: number;

abstract class Attachment { name: string; }

class RegionAttachment extends Attachment {
  width: number; height: number;

  // Applies slot.bone's world transform; NO constraint logic here — constraints are
  // resolved earlier via skeleton.updateWorldTransform().
  computeWorldVertices(slot: Slot, worldVertices: NumberArrayLike, offset: number, stride: number): void;
}

abstract class VertexAttachment extends Attachment {
  worldVerticesLength: number;   // = numVertices * 2 (x,y pairs)
  bones: number[] | null;        // non-null = weighted mesh; null = single-bone mesh

  // For single-bone, applies slot.bone transform.
  // Also applies slot.deform offsets before bone transform (DeformTimeline).
  computeWorldVertices(
    slot: Slot,
    start: number,
    count: number,
    worldVertices: NumberArrayLike,
    offset: number,
    stride: number
  ): void;
}

class PathAttachment extends VertexAttachment { /* skipped */ }
class PointAttachment extends Attachment { /* skipped */ }
```

    ```ts
    import {
      RegionAttachment,
      MeshAttachment,
      VertexAttachment,
      BoundingBoxAttachment,
      PathAttachment,
      PointAttachment,
      ClippingAttachment,
      type Slot,
      type Attachment,
    } from '@esotericsoftware/spine-core';
    import type { AABB, SourceDims } from './types.js';

    /**
     * Compute the world-space AABB of a single attachment on a slot.
     *
     * PRECONDITION (caller responsibility): `skeleton.updateWorldTransform(Physics.update)`
     * has been called since the last bone/constraint mutation. This function does NOT
     * re-run the transform pipeline; it only reads the post-transform bone state via
     * spine-core's computeWorldVertices.
     *
     * Returns null for non-textured attachments (BoundingBox / Path / Point / Clipping).
     */
    export function attachmentWorldAABB(slot: Slot, attachment: Attachment): AABB | null {
      // Order matters: MeshAttachment extends VertexAttachment, so check Region first,
      // then the "skip" types explicitly, then fall through to VertexAttachment.
      if (attachment instanceof RegionAttachment) {
        const v = new Float32Array(8); // 4 vertices × 2 floats
        attachment.computeWorldVertices(slot, v, 0, 2);
        return aabbFromFloat32(v, 4);
      }

      if (
        attachment instanceof BoundingBoxAttachment ||
        attachment instanceof PathAttachment ||
        attachment instanceof PointAttachment ||
        attachment instanceof ClippingAttachment
      ) {
        return null;
      }

      if (attachment instanceof VertexAttachment) {
        // Covers MeshAttachment and any future VertexAttachment subtype with pixels.
        // But we've already filtered out the non-textured VertexAttachment subtypes above.
        const n = attachment.worldVerticesLength;
        if (n <= 0) return null;
        const v = new Float32Array(n);
        attachment.computeWorldVertices(slot, 0, n, v, 0, 2);
        return aabbFromFloat32(v, n / 2);
      }

      return null;
    }

    /**
     * Build an AABB from a flat Float32Array of (x, y) pairs.
     * `vertexCount` is the number of (x, y) pairs, so `buf.length === vertexCount * 2`.
     */
    function aabbFromFloat32(buf: Float32Array, vertexCount: number): AABB {
      let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
      for (let i = 0; i < vertexCount; i++) {
        const x = buf[i * 2];
        const y = buf[i * 2 + 1];
        if (x < minX) minX = x;
        if (x > maxX) maxX = x;
        if (y < minY) minY = y;
        if (y > maxY) maxY = y;
      }
      return { minX, minY, maxX, maxY };
    }

    /**
     * Derive scaleX, scaleY, scale = max(scaleX, scaleY) from an AABB and source dimensions.
     * Per F2.5 all three are reported.
     */
    export function computeScale(
      aabb: AABB,
      sourceDims: SourceDims,
    ): { scaleX: number; scaleY: number; scale: number } {
      const worldW = aabb.maxX - aabb.minX;
      const worldH = aabb.maxY - aabb.minY;
      const scaleX = sourceDims.w > 0 ? worldW / sourceDims.w : 0;
      const scaleY = sourceDims.h > 0 ? worldH / sourceDims.h : 0;
      const scale = Math.max(scaleX, scaleY);
      return { scaleX, scaleY, scale };
    }
    ```

    Notes:
    - Order of `instanceof` checks: `RegionAttachment` first (most common), then the skip set, then `VertexAttachment` as the catch-all for mesh-like attachments. This ordering is deliberate — `MeshAttachment extends VertexAttachment` so the skip types (which also extend VertexAttachment) must be filtered out BEFORE the generic VertexAttachment case.
    - Do NOT reuse Float32Arrays across calls in this module — the sampler will add its own allocation strategy later (plan 04) if it matters for perf. Keep bounds.ts simple and stateless.
