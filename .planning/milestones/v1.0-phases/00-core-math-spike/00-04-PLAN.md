Also run a setup-pose pass so attachments never touched by any animation still get reported.

    ```ts
    import {
      Skeleton,
      AnimationState,
      AnimationStateData,
      Physics,
      type Animation,
      type Skin,
      type SkeletonData,
    } from '@esotericsoftware/spine-core';
    import type { LoadResult, SampleRecord, SourceDims } from './types.js';
    import { attachmentWorldAABB, computeScale } from './bounds.js';

    export const DEFAULT_SAMPLING_HZ = 120;
    const SETUP_POSE_LABEL = 'Setup Pose (Default)';

    export interface SamplerOptions {
      /** Sampling frequency in Hz. Default 120. dt = 1 / samplingHz. */
      samplingHz?: number;
      /**
       * Informational-only "frame" is round(time * frameRate). Default 60.
       * Per CLAUDE.md rule #1 + REQUIREMENTS.md F2.6 — Spine animations are in
       * seconds, this is a display convenience for the UI/CLI.
       */
      frameRate?: number;
    }

    /**
     * Peak record per (skin, slot, attachment) tuple.
     * Keyed in the returned map as `${skinName}/${slotName}/${attachmentName}`.
     */
    export interface PeakRecord extends SampleRecord {
      /** True if no animation timeline touched this attachment; scale is from the setup pose. */
      isSetupPosePeak: boolean;
    }

    /**
     * Run the sampler across every (skin, animation) pair in the loaded skeleton.
     * Returns a map of per-attachment peak records.
     *
     * Locked lifecycle per .planning/phases/00-core-math-spike/00-CONTEXT.md:
     *   For each skin:
     *     skeleton.setSkin(skin); skeleton.setSlotsToSetupPose();
     *     1. Setup-pose pass: setToSetupPose; updateWorldTransform(pose); snapshot.
     *     2. For each animation:
     *        setToSetupPose; state.clearTracks; state.setAnimation(0, anim, false);
     *        updateWorldTransform(Physics.reset);
     *        for t = 0; t <= animation.duration; t += dt:
     *          state.update(dt); state.apply(skeleton);
     *          skeleton.update(dt); skeleton.updateWorldTransform(Physics.update);
     *          snapshot every visible attachment.
     */
    export function sampleSkeleton(
      load: LoadResult,
      opts: SamplerOptions = {},
    ): Map<string, PeakRecord> {
      const samplingHz = opts.samplingHz ?? DEFAULT_SAMPLING_HZ;
      const frameRate = opts.frameRate ?? 60;
      const dt = 1 / samplingHz;

      const peaks = new Map<string, PeakRecord>();
      const touchedByAnimation = new Set<string>(); // keys seen during any animation tick

      const skeleton = new Skeleton(load.skeletonData);
      const stateData = new AnimationStateData(load.skeletonData);
      const state = new AnimationState(stateData);

      for (const skin of load.skeletonData.skins) {
        skeleton.setSkin(skin);
        skeleton.setSlotsToSetupPose();

        // Pass 1: setup-pose snapshot (per skin).
        skeleton.setToSetupPose();
        state.clearTracks();
        skeleton.updateWorldTransform(Physics.pose);
        snapshotFrame(
          skeleton, skin.name, SETUP_POSE_LABEL,
          0, 0, load.sourceDims, peaks, null, frameRate,
        );

        // Pass 2: every animation.
        for (const anim of load.skeletonData.animations) {
          skeleton.setToSetupPose();
          skeleton.setSlotsToSetupPose();
          state.clearTracks();
          state.setAnimation(0, anim, false);
          skeleton.updateWorldTransform(Physics.reset);

          const duration = anim.duration;
          // Inclusive of t=0 and t=duration to catch endpoint peaks.
          for (let t = 0; t <= duration + 1e-9; t += dt) {
            state.update(dt);
            state.apply(skeleton);
            skeleton.update(dt);
            skeleton.updateWorldTransform(Physics.update);
            snapshotFrame(
              skeleton, skin.name, anim.name,
              t, Math.round(t * frameRate),
              load.sourceDims, peaks, touchedByAnimation, frameRate,
            );
          }
        }
      }

      // Mark which peaks are setup-pose-only (never touched by any animation).
      for (const [key, rec] of peaks) {
        if (!touchedByAnimation.has(key) && rec.animationName === SETUP_POSE_LABEL) {
          rec.isSetupPosePeak = true;
        }
      }

      return peaks;
    }

    /**
     * Snapshot every visible attachment on every slot. Update peaks in place.
     * If `touchedSet` is non-null, each visible key is added to it (animation pass).
     */
    function snapshotFrame(
      skeleton: Skeleton,
      skinName: string,
      animationName: string,
      time: number,
      frame: number,
      sourceDims: Map<string, SourceDims>,
      peaks: Map<string, PeakRecord>,
      touchedSet: Set<string> | null,
      _frameRate: number,
    ): void {
      for (const slot of skeleton.slots) {
        const attachment = slot.getAttachment();
        if (attachment === null) continue;
        // Canonical visibility check per CONTEXT.md:
        if (slot.color.a <= 0) continue;

        const aabb = attachmentWorldAABB(slot, attachment);
        if (aabb === null) continue; // BoundingBox/Path/Point/Clipping

        const sd = sourceDims.get(attachment.name);
        if (sd === undefined || sd.w <= 0 || sd.h <= 0) continue;

        const { scaleX, scaleY, scale } = computeScale(aabb, sd);
        const worldW = aabb.maxX - aabb.minX;
        const worldH = aabb.maxY - aabb.minY;

        const key = `${skinName}/${slot.data.name}/${attachment.name}`;
        if (touchedSet !== null) touchedSet.add(key);

        const existing = peaks.get(key);
        if (existing === undefined || scale > existing.scale) {
          peaks.set(key, {
            attachmentKey: key,
            skinName,
            slotName: slot.data.name,
            attachmentName: attachment.name,
            animationName,
            time,
            frame,
            scaleX, scaleY, scale,
            worldW, worldH,
            sourceW: sd.w, sourceH: sd.h,
            isSetupPosePeak: animationName === SETUP_POSE_LABEL,
          });
        }
      }
    }
    ```

    Implementation notes:
    - The setup-pose pass uses `Physics.pose` (sets world transforms without stepping physics) rather than `Physics.update` — setup pose should be static.
    - The tick loop includes `t <= duration + 1e-9` to capture any terminal-frame peak. `animation.duration` can be zero (static poses) — the loop's first iteration still executes because `0 <= 0 + 1e-9`.
    - `Physics.reset` is called ONCE per animation (before the loop), `Physics.update` every tick. This is what guarantees N1.6 determinism.
    - Per-attachment peak is `max(scale)` across all samples; the record keeps the scaleX / scaleY that occurred AT the peak instant (F2.5 — all three reported).
    - Visibility check is `slot.color.a > 0 && slot.getAttachment() !== null`. CONTEXT.md calls this "canonical" — may be tuned if a fixture test uncovers an edge case.
    - The `frameRate` parameter is informational only (F2.6 frame number display). It is NOT multiplied into any math. Per CLAUDE.md rule #1, `skeleton.fps` is not read at all — we use our own `frameRate` (default 60).
