<objective>
Build the headless CLI. Given `npm run cli -- <path-to-skeleton.json>`, it loads the skeleton, runs the sampler, and prints a screenshot-1–style plain-text table to stdout. On a loader error, it prints the typed error's `.message` to stderr and exits non-zero. On success, it prints a footer with the sampler's elapsed time.

Purpose: this is the human-readable verification surface for Phase 0's exit criteria. The user runs this command and eyeballs the table before green-lighting Phase 1. The CLI is also the only deliverable besides tests that proves the end-to-end loader → sampler → report pipeline works.

    ```ts
    #!/usr/bin/env -S tsx
    /**
     * Spine Texture Manager — Phase 0 CLI.
     *
     * Invocation:
     *   npm run cli -- <path/to/skeleton.json> [--hz 120]
     *
     * Prints a plain-text table with columns:
     *   Attachment | Skin | Source W×H | Peak W×H | Scale | Source Animation | Frame
     *
     * Exits 0 on success, non-zero on any loader error.
     */
    import { loadSkeleton } from '../src/core/loader.js';
    import { sampleSkeleton, DEFAULT_SAMPLING_HZ, type PeakRecord } from '../src/core/sampler.js';
    import { SpineLoaderError } from '../src/core/errors.js';

    interface Args {
      skeletonPath: string;
      samplingHz: number;
    }

    function parseArgs(argv: string[]): Args {
      // argv[0] = node binary, argv[1] = script path.
      // When invoked via `tsx scripts/cli.ts -- foo.json`, positional args begin at argv[2].
      const positional: string[] = [];
      let samplingHz = DEFAULT_SAMPLING_HZ;
      for (let i = 2; i < argv.length; i++) {
        const a = argv[i];
        if (a === '--hz' || a === '--samplingHz') {
          const v = argv[++i];
          const n = Number(v);
          if (!Number.isFinite(n) || n <= 0) {
            throw new Error(`Invalid --hz value: ${v}`);
          }
          samplingHz = n;
        } else if (a === '--help' || a === '-h') {
          console.log('Usage: npm run cli -- <path/to/skeleton.json> [--hz 120]');
          process.exit(0);
        } else if (!a.startsWith('-')) {
          positional.push(a);
        } else {
          throw new Error(`Unknown flag: ${a}`);
        }
      }
      if (positional.length !== 1) {
        throw new Error('Usage: npm run cli -- <path/to/skeleton.json> [--hz 120]');
      }
      return { skeletonPath: positional[0], samplingHz };
    }

    function renderTable(peaks: Map<string, PeakRecord>): string {
      const rows: string[][] = [];
      rows.push([
        'Attachment', 'Skin', 'Source W×H', 'Peak W×H', 'Scale', 'Source Animation', 'Frame',
      ]);
      const sorted = [...peaks.values()].sort((a, b) => {
        if (a.skinName !== b.skinName) return a.skinName.localeCompare(b.skinName);
        if (a.slotName !== b.slotName) return a.slotName.localeCompare(b.slotName);
        return a.attachmentName.localeCompare(b.attachmentName);
      });
      for (const rec of sorted) {
        const worldW = rec.worldW.toFixed(1);
        const worldH = rec.worldH.toFixed(1);
        rows.push([
          `${rec.slotName}/${rec.attachmentName}`,
          rec.skinName,
          `${rec.sourceW}×${rec.sourceH}`,
          `${worldW}×${worldH}`,
          rec.scale.toFixed(3),
          rec.animationName,
          String(rec.frame),
        ]);
      }

      // Compute column widths.
      const cols = rows[0].length;
      const widths = new Array<number>(cols).fill(0);
      for (const r of rows) {
        for (let c = 0; c < cols; c++) {
          if (r[c].length > widths[c]) widths[c] = r[c].length;
        }
      }

      const pad = (s: string, w: number) => s + ' '.repeat(w - s.length);
      const out: string[] = [];
      out.push(rows[0].map((s, i) => pad(s, widths[i])).join('  '));
      out.push(widths.map(w => '-'.repeat(w)).join('  '));
      for (let i = 1; i < rows.length; i++) {
        out.push(rows[i].map((s, j) => pad(s, widths[j])).join('  '));
      }
      return out.join('\n');
    }

    function main(): void {
      let args: Args;
      try {
        args = parseArgs(process.argv);
      } catch (e) {
        process.stderr.write(`${(e as Error).message}\n`);
        process.exit(2);
      }

      try {
        const load = loadSkeleton(args.skeletonPath);
        const t0 = performance.now();
        const peaks = sampleSkeleton(load, { samplingHz: args.samplingHz });
        const elapsed = performance.now() - t0;

        process.stdout.write(renderTable(peaks) + '\n');
        process.stdout.write(
          `\nSampled in ${elapsed.toFixed(1)} ms at ${args.samplingHz} Hz ` +
          `(${peaks.size} attachments across ${load.skeletonData.skins.length} skins, ` +
          `${load.skeletonData.animations.length} animations)\n`,
        );
        process.exit(0);
      } catch (e) {
        if (e instanceof SpineLoaderError) {
          process.stderr.write(`${e.name}: ${e.message}\n`);
          process.exit(3);
        }
        process.stderr.write(`Unexpected error: ${(e as Error).stack ?? String(e)}\n`);
        process.exit(1);
      }
    }

    main();
    ```

    Notes:
    - The `#!/usr/bin/env -S tsx` shebang is informational — the script is always run through `tsx` via `npm run cli`. It doesn't need to be executable.
    - Exit codes: 0 success, 2 bad args, 3 loader error, 1 unexpected. CONTEXT.md only requires zero/non-zero; finer codes are the planner's discretion for CI-friendliness.
    - When `npm run cli -- fixtures/.../SIMPLE_TEST.json` is invoked, npm forwards everything after `--` to the script; `tsx` in turn sets `process.argv[2]` to the fixture path. `parseArgs` starts at `argv[2]` accordingly.
    - Table rendering is hand-rolled (per CONTEXT.md: "hand-rolled preferred to avoid dependency bloat").
    - Column separator: two spaces, not pipes. Keeps output easy to diff with the smoke checker in plan 07.

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-00-06-01 | I (Info disclosure) | error stack in stderr | mitigate | Typed `SpineLoaderError` prints only `.name` + `.message` — no stack. Only "unexpected" (last-resort) fallback prints the full stack; acceptable for a dev CLI. |
| T-00-06-02 | D (DoS) | `--hz` accepts huge values | mitigate | `parseArgs` requires `Number.isFinite(n) && n > 0`. An attacker-supplied `--hz 1e9` would produce dt = 1e-9 and loop effectively forever — but this is a local CLI the user runs on themselves. Acceptable; no ceiling added in Phase 0. |
| T-00-06-03 | E (Elevation) | shebang execution | accept | Shebang is informational; the script is run via `tsx` (npm script), not directly. |
