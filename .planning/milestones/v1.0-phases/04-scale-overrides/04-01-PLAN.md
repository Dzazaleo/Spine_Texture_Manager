```typescript
/**
 * Clamp a user-supplied percentage into the valid integer range [1, 100].
 * Non-finite, <1, or non-integer inputs snap to 1; values > 100 snap to 100
 * (F5.2 source-max clamp). Integer-only per D-78.
 */
export function clampOverride(percent: number): number;

/**
 * Apply an override percentage to a peak scale value.
 * Returns { effectiveScale, clamped } — clamped === true iff the raw input
 * (pre-clamp) exceeded 100. The effectiveScale is always computed from the
 * CLAMPED value (peakScale × clampOverride(overridePercent) / 100).
 * Per D-84 the badge still renders when clamped===true.
 */
export function applyOverride(
  peakScale: number,
  overridePercent: number,
): { effectiveScale: number; clamped: boolean };
```

    ```typescript
    export function clampOverride(percent: number): number {
      if (!Number.isFinite(percent)) return 1;
      const int = Math.round(percent);
      if (int < 1) return 1;
      if (int > 100) return 100;
      return int;
    }

    export function applyOverride(
      peakScale: number,
      overridePercent: number,
    ): { effectiveScale: number; clamped: boolean } {
      const clamped = overridePercent > 100;
      const safe = clampOverride(overridePercent);
      return { effectiveScale: peakScale * safe / 100, clamped };
    }
    ```
