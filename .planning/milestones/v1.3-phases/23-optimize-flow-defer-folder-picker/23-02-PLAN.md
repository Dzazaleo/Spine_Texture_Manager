```typescript
// @vitest-environment jsdom
/**
 * Phase 23 — appshell-optimize-flow.spec.tsx
 * Regression tests for the deferred-folder-picker optimize flow (OPT-01, OPT-02).
 *
 * Tests:
 *   (1) D-01: pre-flight header shows "N images" (no path) when outDir is null
 *   (2) D-02: pre-flight header shows "N images → /path" when outDir is set
 *   (3) source-grep gate: onClickOptimize does NOT call pickOutputDirectory
 *   (4) source-grep gate: onConfirmStart DOES call pickOutputDirectory
 *   (5) source-grep gate: buildSessionState does NOT hardcode lastOutDir: null
 *   (6) source-grep gate: onRunEnd contains saveProject call (silent auto-save)
 *
 * Per project test convention (tests/renderer/optimize-dialog-passthrough.spec.tsx:18-19):
 * use `not.toBeNull()` / `toBeDefined()` rather than @testing-library/jest-dom
 * matchers — no jest-dom imports anywhere in tests/renderer.
 */
import { readFile } from 'node:fs/promises';
import { resolve } from 'node:path';
import { afterEach, beforeEach, describe, expect, it, vi } from 'vitest';
import { cleanup, render, screen } from '@testing-library/react';
import { OptimizeDialog } from '../../src/renderer/src/modals/OptimizeDialog';
import type { ExportPlan, ExportRow, PassthroughRow } from '../../src/shared/types';

// ---------------------------------------------------------------------------
// window.api stub — pre-flight state doesn't fire onExportProgress, but the
// useEffect setup still touches the surface, so guard with no-op unsubs.
// ---------------------------------------------------------------------------

beforeEach(() => {
  vi.stubGlobal('api', {
    onExportProgress: vi.fn(() => () => undefined),
    startExport: vi.fn(),
    cancelExport: vi.fn(),
    openOutputFolder: vi.fn(),
    pickOutputDirectory: vi.fn(),
    probeExportConflicts: vi.fn(),
  });
});

afterEach(() => {
  vi.unstubAllGlobals();
  cleanup();
});

// ---------------------------------------------------------------------------
// Factory helpers
// ---------------------------------------------------------------------------

function makeRow(overrides: Partial<ExportRow> = {}): ExportRow {
  return {
    attachmentName: 'test-attachment',
    atlasPage: 'atlas0',
    sourceW: 100,
    sourceH: 100,
    outW: 80,
    outH: 80,
    maxScale: 1,
    outPath: '/out/test-attachment.png',
    sourcePath: '/src/test-attachment.png',
    ...overrides,
  } as ExportRow;
}

function makePassthroughRow(overrides: Partial<PassthroughRow> = {}): PassthroughRow {
  return {
    attachmentName: 'passthrough-attachment',
    atlasPage: 'atlas0',

    outPath: '/out/passthrough-attachment.png',
    sourcePath: '/src/passthrough-attachment.png',
    ...overrides,
  } as PassthroughRow;
}

function makePlan(overrides: Partial<ExportPlan> = {}): ExportPlan {
  return {
    rows: [makeRow()],
    passthroughCopies: [],
    ...overrides,
  } as ExportPlan;
}

// Source-grep regression gates (AppShell.tsx structural contracts)
//
// These tests read the AppShell source directly and verify the Phase 23
// restructure is in place. They act as living documentation of OPT-01/OPT-02.
// ---------------------------------------------------------------------------

describe('AppShell.tsx source-grep gates — Phase 23 OPT-01/OPT-02', () => {
  async function readAppShellSource(): Promise<string> {
    const srcPath = resolve(
      __dirname,
      '..',
      '..',
      'src',
      'renderer',
      'src',
      'components',
      'AppShell.tsx',
    );
    const raw = await readFile(srcPath, 'utf8');
    // Strip block comments then line comments (coarse approach — same as
    // app-shell-output-picker.spec.tsx; avoids false positives in JSDoc).
    return raw
      .replace(/\/\*[\s\S]*?\*\//g, '')
      .split('\n')
      .map((line) => line.replace(/\/\/.*$/, ''))
      .join('\n');
  }

  it('OPT-01 gate: onClickOptimize does NOT call pickOutputDirectory (picker deferred to Start)', async () => {
    const src = await readAppShellSource();
    const match = src.match(/const onClickOptimize\s*=\s*useCallback[\s\S]*?\}\s*,\s*\[/);
    // If the function is found, it must not contain a pickOutputDirectory call.
    if (match) {
      expect(match[0]).not.toMatch(/pickOutputDirectory/);
    }
    // The function must still exist (not deleted).
    expect(src).toMatch(/const onClickOptimize/);
  });

  it('OPT-02 gate: onConfirmStart DOES call pickOutputDirectory (picker moved to Start)', async () => {
    const src = await readAppShellSource();
    const match = src.match(/const onConfirmStart\s*=\s*useCallback[\s\S]*?\}\s*,\s*\[/);
    expect(match).not.toBeNull();
    if (match) {
      expect(match[0]).toMatch(/pickOutputDir/);
    }
  });

  it('D-07 gate: buildSessionState does NOT hardcode lastOutDir: null (real state slot used)', async () => {
    const src = await readAppShellSource();
    // The old hardcoded null must be gone.
    // We look for the pattern "lastOutDir: null" with optional whitespace — a hardcoded null.
    // The new code is just `lastOutDir,` (shorthand property).
    expect(src).not.toMatch(/lastOutDir\s*:\s*null\s*[,}]/);
    // The real state slot declaration must exist.
    expect(src).toMatch(/const \[lastOutDir, setLastOutDir\]/);
  });

  it('D-07 gate: onRunEnd contains silent saveProject call for lastOutDir persistence', async () => {
    const src = await readAppShellSource();
    expect(src).toMatch(/saveProject\(buildSessionState\(\)/);
  });
});
```
